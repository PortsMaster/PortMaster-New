#!/bin/bash
# HELP: PortMaster
# ICON: portmaster
# GRID: PortMaster

. /opt/muos/script/var/func.sh

echo app >/tmp/act_go

GOV_GO="/tmp/gov_go"
[ -e "$GOV_GO" ] && cat "$GOV_GO" >"$(GET_VAR "device" "cpu/governor")"


SETUP_SDL_ENVIRONMENT

HOME="$(GET_VAR "device" "board/home")"
export HOME

SET_VAR "system" "foreground_process" "portmaster"

# Get the script's basename without the path
APP_PATH=$(dirname "$0")
APP_NAME=$(basename "$APP_PATH")
PM_SCRIPT_DIR="/mnt/mmc/ROMS/ports"

if [ ! -f "${PM_SCRIPT_DIR}/${APP_NAME}.sh" ]; then
  PM_SCRIPT_DIR="/mnt/sdcard/ROMS/ports"
fi

"${PM_SCRIPT_DIR}/${APP_NAME}.sh"


unset SDL_ASSERT SDL_HQ_SCALER SDL_ROTATION SDL_BLITTER_DISABLED
